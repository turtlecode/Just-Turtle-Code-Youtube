app.py:


from flask import Flask, request

app = Flask(__name__)

@app.route('/')
def home():
    # We will log the headers to see how Nginx modifies the request
    client_ip = request.headers.get('X-Forwarded-For', request.remote_addr)
    
    return f"""
    <h1>Hello from Python!</h1>
    <p>This response was generated by the Python app running in VS Code.</p>
    <p><b>Your IP Address (seen by Python):</b> {client_ip}</p>
    """

if __name__ == '__main__':
    # Running on port 5000 locally
    print("Starting Python application on http://localhost:5000")
    app.run(host='127.0.0.1', port=5000)


docker-compose.yml:

version: '3.8'

services:
  nginx-proxy:
    image: nginx:latest
    container_name: demo_nginx
    ports:
      # This exposes port 8080 to the entire local network (0.0.0.0:8080)
      - "8080:80"
    volumes:
      - ./nginx.conf:/etc/nginx/nginx.conf:ro
    extra_hosts:
      - "host.docker.internal:host-gateway"

nginx.conf:

events {
    worker_connections 1024;
}

http {
    server {
        listen 80;
        server_name localhost;

        location / {
            # Nginx bypasses external network and talks directly to host's localhost
            proxy_pass http://host.docker.internal:5000;
            
            proxy_set_header Host $host;
            proxy_set_header X-Real-IP $remote_addr;
        }
    }
}
