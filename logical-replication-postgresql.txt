version: '3.8'

services:
  hq_postgres:
    image: postgres:15
    container_name: hq_node
    environment:
      POSTGRES_USER: admin_user
      POSTGRES_PASSWORD: SecretPassword123
      POSTGRES_DB: hq_database
    ports:
      - "5432:5432"
    # Crucial step: Enable logical replication in the Write-Ahead Log
    command: postgres -c wal_level=logical

  branch_postgres:
    image: postgres:15
    container_name: branch_node
    environment:
      POSTGRES_USER: admin_user
      POSTGRES_PASSWORD: SecretPassword123
      POSTGRES_DB: branch_database
    ports:
      - "5433:5432"

  pgadmin:
    image: dpage/pgadmin4
    container_name: pgadmin_monitor
    environment:
      PGADMIN_DEFAULT_EMAIL: admin@demo.com
      PGADMIN_DEFAULT_PASSWORD: admin
    ports:
      - "5050:80"
